-- 转化漏斗分析
SELECT behavior,
  COUNT(*) AS 次数,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 2) AS 占比
FROM user_behavior
GROUP BY behavior
ORDER BY 次数 DESC;
🎯
核心发现：加购→购买是最大流失节点
7,398条加购行为中仅2,902条转化为购买，流失率高达60.78%。用户加购后大量放弃，说明价格敏感度高或存在比价行为。
💡
运营建议：加购用户定向促销
对加购未购买用户推送限时优惠或补贴，预计可将加购→购买转化率从39%提升至55%+，对应GMV增量可观。
Analysis 02 · Repurchase
高复购品类分析
哪些品类拥有最强的用户粘性？30天内同品类购买2次以上的用户数。
Top 10 高复购品类（复购用户数）
同一用户对同一品类产生2次以上购买行为
-- 复购品类分析
SELECT category_id, COUNT(DISTINCT user_id) AS 复购用户数
FROM user_behavior
WHERE behavior = 'buy'
GROUP BY category_id
HAVING COUNT(*) >= 2
ORDER BY 复购用户数 DESC
LIMIT 10;

Analysis 03 · Timing
下单时段分布
用户在一天中的哪些时段最活跃？对运营排期有何指导意义？
24小时下单量分布
以小时为单位统计购买行为次数，识别高峰与低谷
-- 时段分析（需先添加时间字段）
ALTER TABLE user_behavior ADD COLUMN behavior_time DATETIME;
UPDATE user_behavior SET behavior_time = FROM_UNIXTIME(ts);

SELECT HOUR(behavior_time) AS 小时, COUNT(*) AS 下单量
FROM user_behavior WHERE behavior = 'buy'
GROUP BY HOUR(behavior_time) ORDER BY 小时;
